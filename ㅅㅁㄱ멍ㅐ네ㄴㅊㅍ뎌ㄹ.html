func (d *mapDir) ReadDir(count int) ([]fs.DirEntry, error) {
    n := len(d.entry) - d.offset
    if n == 0 && count > 0 {
        return nil, io.EOF
    }
    if count > 0 && n > count {
        n = count
    }
    list := make([]fs.DirEntry, n)
    for i := range list {
        list[i] = &d.entry[d.offset+i]
    }
    d.offset += n
    return list, nil
}
